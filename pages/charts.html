  --create-namespace

üìä Monitoring Setup

Full observability with Prometheus and Grafana

# values-monitoring.yaml
metrics:
  enabled: true
  serviceMonitor:
    enabled: true
    interval: 30s
    labels:
      monitoring: prometheus

observability:
  enabled: true
  grafana:
    enabled: true
    dashboards:
      - security-overview
      - attack-surface
      - policy-violations

prometheusRule:
  enabled: true
  rules:
    - alert: PahlevanDown
      expr: up{job="pahlevan-operator"} == 0
      for: 5m

üß™ Testing Charts

Local development and testing workflow:

# Clone repository
git clone https://github.com/obsernetics/pahlevan.git
cd pahlevan

# Test chart template
helm template test charts/pahlevan-operator \
  --values charts/pahlevan-operator/values.yaml

helm install pahlevan charts/pahlevan-operator \
  --namespace pahlevan-system \
  --create-namespace \
  --dry-run

# Run chart tests
helm test pahlevan -n pahlevan-system
