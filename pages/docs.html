🚀 Installation

# Helm installation
helm repo add pahlevan https://obsernetics.github.io/pahlevan/
helm install pahlevan pahlevan/pahlevan-operator \
  --namespace pahlevan-system \
  --create-namespace

# Direct installation
kubectl apply -f https://github.com/obsernetics/pahlevan/releases/latest/download/install.yaml

📊 Status Check

# Check operator status
kubectl get pods -n pahlevan-system

# View security policies
kubectl get securitypolicies -A

# Check eBPF programs
kubectl logs -n pahlevan-system deployment/pahlevan-operator

🔧 Configuration

# Basic policy example
apiVersion: security.pahlevan.io/v1alpha1
kind: SecurityPolicy
metadata:
  name: web-app-policy
spec:
  selector:
    matchLabels:
      tier: frontend
  learningDuration: "5m"
  enforcementMode: "Monitor"

📈 Monitoring

# Access metrics
kubectl port-forward -n pahlevan-system svc/pahlevan-operator 8080:8080
curl http://localhost:8080/metrics

# View events
kubectl get events -n pahlevan-system --sort-by='.lastTimestamp'
